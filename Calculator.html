const sheetURL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vSgsCQ59YbAExP9Ik6g_vLI3WL2eIOGmY3D48S28e44cRTJjr2xwEpjKSQ4Z0GCK8q6Q9H5WStT0Xa4/pub?gid=319738758&single=true&output=csv"; let globalData = []; async function loadData() { const margin = parseFloat(document.getElementById("margin").value) || 0; let res = await fetch(sheetURL); let text = await res.text(); let rows = text.split("\n").map(r => r.split(",")); let headers = rows[0]; globalData = rows.slice(1).map(r => { let obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); renderTable(globalData, margin); } function calculate(row, margin) { let TP = parseFloat(row.TP || 0); let SP = TP * (1 + margin / 100); let Commission = SP * 0.37; let GST = Commission * 0.18; let TDS = SP * 0.001; let Gross = SP - Commission - GST - TDS; let Processing = 99 * 1.18; let Dispatch = SP < 500 ? 25 : SP < 1000 ? 30 : 35; let Net = Gross - Processing - Dispatch; let diff = Net - TP; let diffPer = diff / TP; let MRP = Math.round((SP / 35) * 100); return { SP, MRP, diffPer }; } function renderTable(data, margin) { let tbody = document.querySelector("#table tbody"); tbody.innerHTML = ""; data.forEach(row => { let calc = calculate(row, margin); let tr = document.createElement("tr"); tr.innerHTML = ` ${row.Product_Name} ${row.Brand} ${row.Status} ${row.TP} ${calc.SP.toFixed(2)} ${calc.MRP} ${(calc.diffPer * 100).toFixed(2)}% `; tbody.appendChild(tr); }); } function filterTable() { let search = document.getElementById("search").value.toLowerCase(); let filtered = globalData.filter(r => r.Product_Name.toLowerCase().includes(search) ); renderTable(filtered, document.getElementById("margin").value); } function downloadCSV() { let rows = [["Product","SP"]]; document.querySelectorAll("#table tbody tr").forEach(tr => { let cols = tr.querySelectorAll("td"); rows.push([cols[0].innerText, cols[4].innerText]); }); let csv = rows.map(r => r.join(",")).join("\n"); let blob = new Blob([csv]); let a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = "pricing.csv"; a.click(); } function saveHistory() { let data = document.querySelector("#table tbody").innerHTML; let history = JSON.parse(localStorage.getItem("history") || "[]"); history.push({ date: new Date().toLocaleString(), data: data }); localStorage.setItem("history", JSON.stringify(history)); alert("Saved!"); }
